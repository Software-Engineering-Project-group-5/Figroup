const express = require('express');
const router = express.Router();
const contributionController = require('../controllers/contributionController');
const auth = require('../middleware/auth');

/**
 * @module ContributionRouter
 * @description Routes for managing contributions
 */
/**
 * @route GET /api/investments/{investment_id}/contributions
 * @group Contributions - Investment contribution management
 * @summary Get all contributions for an investment
 * @description Retrieves a list of all contributions associated with a specific investment
 * @security bearerAuth
 * @param {string} investment_id.path.required - ID of the investment - example: "665f6a9e4d2c1220d4a84b3d"
 * @returns {Array.<object>} 200 - List of contributions retrieved successfully
 * @returns {string} 200._id - Contribution ID
 * @returns {string} 200.investment_id - Associated investment ID
 * @returns {string} 200.user_id - Contributor's user ID
 * @returns {number} 200.amount - Contribution amount
 * @returns {string} 200.group_id - Associated group ID
 * @returns {string} 200.created_at - Creation timestamp (date-time format) - example: "2024-06-05T14:30:00.000Z"
 * @returns {object} 401 - Unauthorized - example: { msg: "Token is not valid" }
 */
router.get('/:investment_id/contributions', auth, contributionController.getContributions);

module.exports = router;
