const express = require('express');
const router = express.Router();
const investmentController = require('../controllers/investmentController');
const auth = require('../middleware/auth');

/**
 * @module StocksRouter
 * @description Routes for managing stocks
 */
/**
 * @route   GET /api/companies
 * @desc    Get all companies
 * @access  Private
 * @security bearerAuth: []
 * @returns {Object} 200 - List of companies with their stock symbols
 * @returns {Object} 401 - Unauthorized - Invalid or expired token
 * @example
 * // Example response for 200
 * {
 *   "success": true,
 *   "count": 103,
 *   "stocks": {
 *     "Apple Inc.": "AAPL",
 *     "Microsoft Corporation": "MSFT",
 *     "Alphabet Inc. (Google)": "GOOGL",
 *     "Amazon.com Inc.": "AMZN",
 *     "NVIDIA Corporation": "NVDA",
 *     "Meta Platforms Inc.": "META"
 *   }
 * }
 * 
 * @example
 * // Example response for 401
 * {
 *   "msg": "Token is not valid"
 * }
 */
router.get('/', auth, investmentController.getCompanyDetails);

module.exports = router;
