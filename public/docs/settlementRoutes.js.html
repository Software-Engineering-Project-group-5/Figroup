const express = require('express');
const router = express.Router();
const settlementController = require('../controllers/settlementController');
const auth = require('../middleware/auth');

/**
 * @module SettlementRouter
 * @description Routes for managing settlements
 */
/**
 * Create a new settlement
 * 
 * @route POST /api/settlements
 * @description Records a new settlement between users to resolve debts or expenses within a group
 * @access Private
 * 
 * @param {Object} req.body.required - Request body
 * @param {string} req.body.group_id - ID of the group where settlement is occurring - eg: "67d5ad9e3cc22e833a11dd32"
 * @param {string} req.body.from_user_id - ID of the user who owes money (payer) - eg: "67d5ace83cc22e833a11dd2d"
 * @param {string} req.body.to_user_id - ID of the user who is owed money (payee) - eg: "67d5a8609335c8e6e12013cc"
 * 
 * @returns {Object} 201 - Settlement created successfully
 * @returns {string} 201.Settlement_id - Unique identifier for the created settlement - eg: "67f2cb453a8921f09ad3fe12"
 * 
 * @returns {Object} 400 - Invalid input
 * @returns {string} 400.msg - Error message - eg: "Invalid settlement data"
 * 
 * @returns {Object} 401 - Unauthorized
 * @returns {string} 401.msg - Error message - eg: "Token is not valid"
 */
router.post('/', auth, settlementController.createSettlement);

/**
 * Get settlement details
 * 
 * @route GET /api/settlements/{settlement_id}
 * @description Retrieves information about a specific settlement by its ID
 * @access Private
 * 
 * @param {string} settlement_id.path.required - Unique identifier of the settlement - eg: "67f2cb453a8921f09ad3fe12"
 * 
 * @returns {Object} 200 - Settlement details retrieved
 * @returns {string} 200._id - Settlement ID - eg: "67f2cb453a8921f09ad3fe12"
 * @returns {string} 200.group_id - ID of the associated group - eg: "67d5ad9e3cc22e833a11dd32"
 * @returns {string} 200.from_user_id - ID of the user who owes money - eg: "67d5ace83cc22e833a11dd2d"
 * @returns {string} 200.to_user_id - ID of the user who is owed money - eg: "67d5a8609335c8e6e12013cc"
 * @returns {string} 200.status - Current status of the settlement (PENDING, COMPLETED) - eg: "PENDING"
 * @returns {string} 200.created_at - Settlement creation timestamp - eg: "2025-04-05T23:45:25.850Z"
 * @returns {string} 200.updated_at - Settlement last update timestamp - eg: "2025-04-05T23:45:25.850Z"
 * 
 * @returns {Object} 404 - Settlement not found
 * @returns {string} 404.msg - Error message - eg: "Settlement not found"
 * 
 * @returns {Object} 401 - Unauthorized
 * @returns {string} 401.msg - Error message - eg: "Token is not valid"
 */
router.get('/:settlement_id', auth, settlementController.getSettlements);

module.exports = router;
